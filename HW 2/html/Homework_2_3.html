% Function Definitions

function [ LP ] = Lagrange ( X, Y )
    t = sym('t');
    order = length(X);
    A = [];
    for i = 1:order
        A = [A ; row(X(i), order - 1)];
    end

    x = A\Y';
    T = row (t, order - 1);
    LP = T * x;

    function [ A ] = row ( x, order )
        A = [];
        for j = 0:order
            A = [A, x^j];
        end
    end
end

function [ A ] = row ( x, order )
    A = [];
    for i = 0:order
        A = [A, x^i];
    end
end
